#fuction to plot boxplot
def boxplot(df,title):
    sns.set(rc = {'figure.figsize':(20,4)})
    return sns.boxplot(df['runtime']).set_title(title)
#fuction to plot barplot
def barplot(data,title,xlabel,ylabel):
    plt.figure(figsize=(20,4),dpi=100)
    plt.bar(data.index,data)
    plt.title(title)
    plt.xlabel(xlabel)
    plt.ylabel(ylabel)
    plt.show()
#function to plot line plot
def lineplot(data,title,xlabel,ylabel):
    plt.figure(figsize=(6,3),dpi=100)
    plt.title(title)
    plt.xlabel(xlabel)
    plt.ylabel(ylabel)
    plt.plot(data.index,data)
    plt.show()
